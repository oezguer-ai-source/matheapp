    import type { Metadata } from "next";
    import { redirect } from "next/navigation";
    import { createClient } from "@/lib/supabase/server";
    import { LogoutButtonChild } from "@/components/child/logout-button";

    export const metadata: Metadata = {
      title: "Matheapp â€” Startseite",
    };

    export default async function KindDashboardPage() {
      const supabase = await createClient();
      const { data: claims } = await supabase.auth.getClaims();
      if (!claims?.claims) {
        redirect("/login");

      const userId = claims.claims.sub;
      const { data: profile } = await supabase

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Middleware -> App routes | Session validity check; any unauthenticated request crosses this boundary |
| JWT claims -> Role enforcement | `app_metadata.role` is the middleware's only source of truth |
| Server Component -> Database | `getClaims()` validates the JWT; profile lookup runs under user-context RLS |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-08-01 | Spoofing | Attacker forges a JWT to impersonate a teacher | mitigate | `supabase.auth.getClaims()` validates the JWT signature against Supabase's key. Unsigned/forged tokens fail validation. |
| T-10-08-02 | Elevation of Privilege | Child bypasses middleware by setting a `/kind/...` URL that reads teacher data | mitigate | Server Component itself calls `supabase.auth.getClaims()` + reads `profile.role` + redirects if role mismatches. Middleware + page-level check are defense in depth. |
| T-10-08-03 | Elevation of Privilege | Middleware matcher accidentally skips Server Actions | mitigate | Matcher pattern is verbatim from RESEARCH Pattern 4; Server Actions POST to the same route pattern and are covered. Pitfall 1 is specifically about this. |
| T-10-08-04 | Information Disclosure | Server Component leaks other users' `display_name` | mitigate | Profile lookup uses user-context `createClient()` (NOT admin); RLS policy `child_reads_own_profile` / `teacher_reads_class_profiles` from Plan 05 scopes the result. |
| T-10-08-05 | Denial of Service | Middleware hits the database on every static asset request | mitigate | Matcher excludes `_next/static`, `_next/image`, and common image extensions. Middleware only runs on app routes + Server Actions. |
</threat_model>
